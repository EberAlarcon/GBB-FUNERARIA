--CONSULTA 1--
SELECT 
CONCAT (NOMBRE_DIFUNTO,' ',APELLIDO_DIFUNTO)NOMBRES_Y_APELLIDOS, 
(HORAEMBALSAMIENTO_DIFUNTO - HORADEFUNSION_DIFUNTO) AS TIEMPO_DEFUNSION_Y_EMBALSMIENTO
FROM 
DIFUNTO

--CONSULTA 2--
SELECT 
NOMBRE_ESPACIO AS NOMBRE_DE_LA_SALA, 
COUNT (*) CANTIDAD_DE_DIFUNTOS, 
SUM (PRECIO_ESPACIO) CANTIDAD_DE_DINERO
FROM 
SALA_VELATORIO
GROUP BY 
NOMBRE_ESPACIO;

--CONSULTA 3--
SELECT 
DESCRIPCION_SERVICIO AS NOMBRE_DEL_SERVICIO, 
COUNT (NOMBRE_DIFUNTO) AS CANTIDAD_DE_DIFUNTOS, 
SUM (COSTO_SERVICIO) AS GANANCIAS_SERVICIOS
FROM 
DIFUNTO  
INNER JOIN FAMILIAR  ON DIFUNTO.ID_FAMLIAR=FAMILIAR.ID_FAMLIAR
INNER JOIN SERVICIO ON FAMILIAR.ID_FAMLIAR=SERVICIO.ID_FAMLIAR
GROUP BY  
DESCRIPCION_SERVICIO;

--CONSULTA 4--
SELECT 
DESCRIPCION_CAUSA, 
COUNT(GENERO_DIFUNTO) AS CANTIDAD_MUJERES, 
COUNT(GENERO_DIFUNTO) AS CANTIDAD_HOMBRES 
FROM
CAUSA 
c INNER JOIN DIFUNTO d ON 
c.ID_DIFUNTO=d.ID_DIFUNTO
WHERE GENERO_DIFUNTO='MUJER' or GENERO_DIFUNTO='hombre'
GROUP BY 
DESCRIPCION_CAUSA;
