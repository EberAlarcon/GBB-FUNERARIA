SELECT 
DESCRIPCION_SERVICIO AS NOMBRE_DEL_SERVICIO, 
costo_servicio,
COUNT (NOMBRE_DIFUNTO) AS CANTIDAD_DE_DIFUNTOS, 
SUM (COSTO_SERVICIO) AS GANANCIAS_SERVICIOS
FROM 
DIFUNTO  
INNER JOIN FAMILIAR  ON DIFUNTO.ID_FAMLIAR=FAMILIAR.ID_FAMLIAR
INNER JOIN SERVICIO ON FAMILIAR.ID_FAMLIAR=SERVICIO.ID_FAMLIAR
GROUP BY  
DESCRIPCION_SERVICIO, costo_servicio
